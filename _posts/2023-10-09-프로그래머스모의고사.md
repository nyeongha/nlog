### 프로래그래스머스 모의고사-python

```python
def solution(answers):
    answer={1:0,2:0,3:0}
    sp1=[1, 2, 3, 4, 5]
    sp2=[2, 1, 2, 3, 2, 4, 2, 5]
    sp3=[3, 3, 1, 1, 2, 2, 4, 4, 5, 5]
    for i in range(len(answers)):
        if sp1[i%5]==answers[i]:
            answer[1]+=1
        if sp2[i%8]==answers[i]:
            answer[2]+=1
        if sp3[i%10]==answers[i]:
            answer[3]+=1
    aa=[]
    for i,j in answer.items():
        if max(answer.values())==j:
            aa.append(i)
    aa.sort()
    return aa
```
